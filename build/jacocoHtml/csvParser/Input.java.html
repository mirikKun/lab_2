package csvParser;

import java.util.Scanner;

public class Input {
    private static Propety ff;

    public static Propety newIn() {
        return getIn(new Asker(System.in, System.out));
    }

    public static Propety getIn(Asker asker)
    {
        Scanner Scan = new Scanner(System.in);
        //System.out.println("Р’РІРµРґРёС‚Рµ РёС‚РѕРіРѕРІС‹Р№ СЃРµРїР°СЂР°С‚РѕСЂ:");
        String plus = asker.ask("Р’РІРµРґРёС‚Рµ РёС‚РѕРіРѕРІС‹Р№ СЃРµРїР°СЂР°С‚РѕСЂ:");
        if (!plus.equals("")) {
            //System.out.println("Р’РІРµРґРёС‚Рµ РїСѓС‚СЊ Рє С„Р°Р№Р»Сѓ:");
            String to = asker.ask("Р’РІРµРґРёС‚Рµ РїСѓС‚СЊ Рє С„Р°Р№Р»Сѓ:");
            if (!to.equals("")){
                //System.out.println("Р’РІРµРґРёС‚Рµ РїСѓС‚СЊ СЃРѕР·РґР°РЅРёСЏ:");
                String from = asker.ask("Р’РІРµРґРёС‚Рµ РїСѓС‚СЊ СЃРѕР·РґР°РЅРёСЏ:");
                if (!from.equals("")){
                    //System.out.println("Р’РІРµРґРёС‚Рµ СЃРµРїР°СЂР°С‚РѕСЂ:");
                    String sep = asker.ask("Р’РІРµРґРёС‚Рµ СЃРµРїР°СЂР°С‚РѕСЂ:");
                    if (!sep.equals(""))
                        ff=new Propety(plus,to,from,sep.charAt(0));
                    else
                        ff=new Propety(plus,to,from);
                }
                else ff=new Propety(plus,to);
            }
            else
                ff=new Propety(plus);
        }
        else{
            ff= new Propety();
        }
        return ff;
    }
}
